 50 -def runpf(casedata='case9', ppopt=None, fname='', solvedcase=''): 

 51      """Runs a power flow. 
 52   
 53      Runs a power flow [full AC Newton's method by default] and optionally 
 54      returns the solved values in the data matrices, a flag which is C{True} if 
 55      the algorithm was successful in finding a solution, and the elapsed 
 56      time in seconds. All input arguments are optional. If C{casename} is 
 57      provided it specifies the name of the input data file or dict 
 58      containing the power flow data. The default value is 'case9'. 
 59   
 60      If the ppopt is provided it overrides the default PYPOWER options 
 61      vector and can be used to specify the solution algorithm and output 
 62      options among other things. If the 3rd argument is given the pretty 
 63      printed output will be appended to the file whose name is given in 
 64      C{fname}. If C{solvedcase} is specified the solved case will be written 
 65      to a case file in PYPOWER format with the specified name. If C{solvedcase} 
 66      ends with '.mat' it saves the case as a MAT-file otherwise it saves it 
 67      as a Python-file. 
 68   
 69      If the C{ENFORCE_Q_LIMS} options is set to C{True} [default is false] then 
 70      if any generator reactive power limit is violated after running the AC 
 71      power flow, the corresponding bus is converted to a PQ bus, with Qg at 
 72      the limit, and the case is re-run. The voltage magnitude at the bus 
 73      will deviate from the specified value in order to satisfy the reactive 
 74      power limit. If the reference bus is converted to PQ, the first 
 75      remaining PV bus will be used as the slack bus for the next iteration. 
 76      This may result in the real power output at this generator being 
 77      slightly off from the specified values. 
 78      """ 
 79      ## default arguments 
 80      ppopt = ppoption(ppopt) 
 81   
 82      ## options 
 83      verbose = ppopt["VERBOSE"] 
 84      qlim = ppopt["ENFORCE_Q_LIMS"]  ## enforce Q limits on gens? 
 85      dc = ppopt["PF_DC"]             ## use DC formulation? 
 86   
 87      ## read data 
 88      ppc = loadcase(casedata) 
 89   
 90      ## add zero columns to branch for flows if needed 
 91      if ppc["branch"].shape[1] < QT: 
 92          ppc["branch"] = c_[ppc["branch"], 
 93                             zeros((ppc["branch"].shape[0], 
 94                                    QT - ppc["branch"].shape[1] + 1))] 
 95   
 96      ## convert to internal indexing 
 97      ppc = ext2int(ppc) 
 98      baseMVA, bus, gen, branch = \ 
 99          ppc["baseMVA"], ppc["bus"], ppc["gen"], ppc["branch"] 
100   
101      ## get bus index lists of each type of bus 
102      ref, pv, pq = bustypes(bus, gen) 
103   
104      ## generator info 
105      on = find(gen[:, GEN_STATUS] > 0)      ## which generators are on? 
106      gbus = gen[on, GEN_BUS].astype(int)    ## what buses are they at? 
107   
108      ##-----  run the power flow  ----- 
109      t0 = time() 
110      if verbose > 0: 
111          v = ppver('all') 
112          stdout.write('PYPOWER Version %s, %s' % (v["Version"], v["Date"])) 
113   
114      if dc:                               # DC formulation 
115          if verbose: 
116              stdout.write(' -- DC Power Flow\n') 
117   
118          ## initial state 
119          Va0 = bus[:, VA] * (pi / 180) 
120   
121          ## build B matrices and phase shift injections 
122          B, Bf, Pbusinj, Pfinj = makeBdc(baseMVA, bus, branch) 
123   
124          ## compute complex bus power injections [generation - load] 
125          ## adjusted for phase shifters and real shunts 
126          Pbus = makeSbus(baseMVA, bus, gen).real - Pbusinj - bus[:, GS] / baseMVA 
127   
128          ## "run" the power flow 
129          Va = dcpf(B, Pbus, Va0, ref, pv, pq) 
130   
131          ## update data matrices with solution 
132          branch[:, [QF, QT]] = zeros((branch.shape[0], 2)) 
133          branch[:, PF] = (Bf * Va + Pfinj) * baseMVA 
134          branch[:, PT] = -branch[:, PF] 
135          bus[:, VM] = ones(bus.shape[0]) 
136          bus[:, VA] = Va * (180 / pi) 
137          ## update Pg for swing generator [note: other gens at ref bus are accounted for in Pbus] 
138          ##      Pg = Pinj + Pload + Gs 
139          ##      newPg = oldPg + newPinj - oldPinj 
140          refgen = find(gbus == ref)             ## which is[are] the reference gen[s]? 
141          gen[on[refgen[0]], PG] = gen[on[refgen[0]], PG] + (B[ref, :] * Va - Pbus[ref]) * baseMVA 
142   
143          success = 1 
144      else:                                ## AC formulation 
145          if verbose > 0: 
146              stdout.write(' -- AC Power Flow ')    ## solver name and \n added later 
147   
148          ## initial state 
149          # V0    = ones(bus.shape[0])            ## flat start 
150          V0  = bus[:, VM] * exp(1j * pi/180 * bus[:, VA]) 
151          V0[gbus] = gen[on, VG] / abs(V0[gbus]) * V0[gbus] 
152   
153          if qlim: 
154              ref0 = ref                         ## save index and angle of 
155              Varef0 = bus[ref0, VA]             ##   original reference bus 
156              limited = []                       ## list of indices of gens @ Q lims 
157              fixedQg = zeros(gen.shape[0])      ## Qg of gens at Q limits 
158   
159          repeat = True 
160          while repeat: 
161              ## build admittance matrices 
162              Ybus, Yf, Yt = makeYbus(baseMVA, bus, branch) 
163   
164              ## compute complex bus power injections [generation - load] 
165              Sbus = makeSbus(baseMVA, bus, gen) 
166   
167              ## run the power flow 
168              alg = ppopt["PF_ALG"] 
169              if alg == 1: 
170                  V, success, _ = newtonpf(Ybus, Sbus, V0, ref, pv, pq, ppopt) 
171              elif alg == 2 or alg == 3: 
172                  Bp, Bpp = makeB(baseMVA, bus, branch, alg) 
173                  V, success, _ = fdpf(Ybus, Sbus, V0, Bp, Bpp, ref, pv, pq, ppopt) 
174              elif alg == 4: 
175                  V, success, _ = gausspf(Ybus, Sbus, V0, ref, pv, pq, ppopt) 
176              else: 
177                  stderr.write('Only Newton''s method, fast-decoupled, and ' 
178                               'Gauss-Seidel power flow algorithms currently ' 
179                               'implemented.\n') 
180   
181              ## update data matrices with solution 
182              bus, gen, branch = pfsoln(baseMVA, bus, gen, branch, Ybus, Yf, Yt, V, ref, pv, pq) 
183   
184              if qlim:             ## enforce generator Q limits 
185                  ## find gens with violated Q constraints 
186                  mx = find( gen[:, GEN_STATUS] > 0 & gen[:, QG] > gen[:, QMAX] ) 
187                  mn = find( gen[:, GEN_STATUS] > 0 & gen[:, QG] < gen[:, QMIN] ) 
188   
189                  if len(mx) > 0 or len(mn) > 0:  ## we have some Q limit violations 
190                      if len(pv): 
191                          if verbose: 
192                              if len(mx) > 0: 
193                                  print 'Gen %d [only one left] exceeds upper Q limit : INFEASIBLE PROBLEM\n' % mx 
194                              else: 
195                                  print 'Gen %d [only one left] exceeds lower Q limit : INFEASIBLE PROBLEM\n' % mn 
196   
197                          success = 0 
198                          break 
199   
200                      ## one at a time? 
201                      if qlim == 2:    ## fix largest violation, ignore the rest 
202                          k = argmax(r_[gen[mx, QG] - gen[mx, QMAX], 
203                                        gen[mn, QMIN] - gen[mn, QG]]) 
204                          if k > len(mx): 
205                              mn = mn[k - len(mx)] 
206                              mx = [] 
207                          else: 
208                              mx = mx[k] 
209                              mn = [] 
210   
211                      if verbose and len(mx) > 0: 
212                          print 'Gen %d at upper Q limit, converting to PQ bus\n' % mx 
213   
214                      if verbose and len(mn) > 0: 
215                          print 'Gen %d at lower Q limit, converting to PQ bus\n' % mn 
216   
217                      ## save corresponding limit values 
218                      fixedQg[mx] = gen[mx, QMAX] 
219                      fixedQg[mn] = gen[mn, QMIN] 
220                      mx = r_[mx, mn] 
221   
222                      ## convert to PQ bus 
223                      gen[mx, QG] = fixedQg[mx]      ## set Qg to binding limit 
224                      gen[mx, GEN_STATUS] = 0        ## temporarily turn off gen, 
225                      for i in range(mx):            ## [one at a time, since they may be at same bus] 
226                          bi = gen[mx[i], GEN_BUS]   ## adjust load accordingly, 
227                          bus[bi, [PD, QD]] = (bus[bi, [PD, QD]] - gen[mx[i], [PG, QG]]) 
228   
229                      bus[gen[mx, GEN_BUS], BUS_TYPE] = PQ   ## & set bus type to PQ 
230   
231                      ## update bus index lists of each type of bus 
232                      ref_temp = ref 
233                      ref, pv, pq = bustypes(bus, gen) 
234                      if verbose and ref != ref_temp: 
235                          print 'Bus %d is new slack bus\n' % ref 
236   
237                      limited = r_[limited, mx] 
238                  else: 
239                      repeat = 0 ## no more generator Q limits violated 
240              else: 
241                  repeat = 0     ## don't enforce generator Q limits, once is enough 
242   
243          if qlim and len(limited) > 0: 
244              ## restore injections from limited gens [those at Q limits] 
245              gen[limited, QG] = fixedQg[limited]    ## restore Qg value, 
246              for i in range(limited):               ## [one at a time, since they may be at same bus] 
247                  bi = gen[limited[i], GEN_BUS]      ## re-adjust load, 
248                  bus[bi, [PD, QD]] = bus[bi, [PD, QD]] + gen[limited[i], [PG, QG]] 
249   
250              gen[limited, GEN_STATUS] = 1           ## and turn gen back on 
251              if ref != ref0: 
252                  ## adjust voltage angles to make original ref bus correct 
253                  bus[:, VA] = bus[:, VA] - bus[ref0, VA] + Varef0 
254   
255      ppc["et"] = time() - t0 
256      ppc["success"] = success 
257   
258      ##-----  output results  ----- 
259      ## convert back to original bus numbering & print results 
260      ppc["bus"], ppc["gen"], ppc["branch"] = bus, gen, branch 
261      results = int2ext(ppc) 
262   
263      ## zero out result fields of out-of-service gens & branches 
264      if len(results["order"]["gen"]["status"]["off"]) > 0: 
265          results["gen"][ix_(results["order"]["gen"]["status"]["off"], [PG, QG])] = 0 
266   
267      if len(results["order"]["branch"]["status"]["off"]) > 0: 
268          results["branch"][ix_(results["order"]["branch"]["status"]["off"], [PF, QF, PT, QT])] = 0 
269   
270      if fname: 
271          fd = None 
272          try: 
273              fd = open(fname, "wb") 
274          except Exception, detail: 
275              stderr.write("Error opening %s: %s.\n" % (fname, detail)) 
276          finally: 
277              if fd is not None: 
278                  printpf(results, fd, ppopt) 
279                  fd.close() 
280   
281      printpf(results, stdout, ppopt) 
282   
283      ## save solved case 
284      if solvedcase: 
285          savecase(solvedcase, results) 
286   
287      return results, success

291      runpf()
